### 5.2 统一错误处理

```javascript
// ❌ 当前：多处静默吞错
try { GM_setValue(...); } catch(e) {}

// ✅ 改进：统一处理器
function safeStorage(action, key, value) {
  try {
    if (action === 'set') GM_setValue(key, JSON.stringify(value));
    if (action === 'get') return GM_getValue(key);
  } catch (e) {
    uLog(`⚠️ 存储${action}失败: ${key}`, 'warn');
    return action === 'get' ? undefined : false;
  }
}
```

### 5.3 提取状态机保存函数

```javascript
// ✅ 将重复4次的逻辑提取为函数
function saveCurrentQA(db, currentQALines, currentAnswer, results) {
  if (!currentAnswer) return;
  const qText = cleanQuestionText(currentQALines.join('\n'));
  if (qText.length < THRESHOLDS.MIN_QUESTION_LEN) { results.skipped++; return; }
  const ans = normalizeAnswer(currentAnswer);
  if (db.hasOwnProperty(qText)) {
    results.duplicates.push({ q: qText, oldAns: db[qText], newAns: ans });
    results.skipped++;
  } else {
    db[qText] = ans;
    results.added++;
  }
}
```
